import { NextRequest, NextResponse } from 'next/server'
import { getServerSession } from 'next-auth'
import { authOptions } from '@/lib/auth-config'
import prisma from '@/lib/prisma'

  request: NextRequest,
  { params }: { params: { id: string } }
) {

        { error: 'Unauthorized' },
        { status: 401 }
      )
    }
